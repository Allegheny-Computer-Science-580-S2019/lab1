    \begin{alertblock}{More precise definition}
    Given two nonnegative functions $f$ and $g$, $f(N)$ is  $O(g(N))$ iff $\exists$ constant $c > 0$ so that $f(N) \leq c g(N)$
    \end{alertblock}
